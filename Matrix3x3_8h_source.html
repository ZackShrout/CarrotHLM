210 inline float3x3 rotate_axis_angle(const float3 axis, const float rad) noexcept
211 {
212 const quat q{ quat_from_axis_angle(axis, rad) };
213
214 const float xx{ q.x * q.x };
215 const float yy{ q.y * q.y };
216 const float zz{ q.z * q.z };
217 const float xy{ q.x * q.y };
218 const float xz{ q.x * q.z };
219 const float yz{ q.y * q.z };
220 const float wx{ q.w * q.x };
221 const float wy{ q.w * q.y };
222 const float wz{ q.w * q.z };
223
224 return {
225 float3{ 1.f - 2.f * (yy + zz), 2.f * (xy - wz), 2.f * (xz + wy) },
226 float3{ 2.f * (xy + wz), 1.f - 2.f * (xx + zz), 2.f * (yz - wx) },
227 float3{ 2.f * (xz - wy), 2.f * (yz + wx), 1.f - 2.f * (xx + yy) }
228 };

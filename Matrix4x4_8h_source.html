182 inline float4 mul(const float4x4& m, const float4& v) noexcept
183 {
184 return v.x * m.columns[0] +
185 v.y * m.columns[1] +
186 v.z * m.columns[2] +
187 v.w * m.columns[3];
188 }
189
197 inline float4x4 mul(const float4x4& a, const float4x4& b) noexcept
198 {
199 float4x4 result;
200 result.columns[0] = mul(a, b.columns[0]);
201 result.columns[1] = mul(a, b.columns[1]);
202 result.columns[2] = mul(a, b.columns[2]);
203 result.columns[3] = mul(a, b.columns[3]);
204 return result;
205 }

280 inline float4x4 float4x4::rotate_axis_angle(float3 axis, const float rad) noexcept
281 {
282 axis = normalize(axis);
283 const float c{ std::cos(rad) };
284 const float s{ std::sin(rad) };
285 const float t{ 1.0f - c };
286
287 const float x{ axis.x };
288 const float y{ axis.y };
289 const float z{ axis.z };
290
291 return float4x4{
292 float4{ t * x * x + c, t * x * y + s * z, t * x * z - s * y, 0.f },
293 float4{ t * x * y - s * z, t * y * y + c, t * y * z + s * x, 0.f },
294 float4{ t * x * z + s * y, t * y * z - s * x, t * z * z + c, 0.f },
295 float4{ 0.f, 0.f, 0.f, 1.f }
296 };

299 inline float4x4 float4x4::look_at_lh(const float3 eye, const float3 target, const float3 up) noexcept
300 {
301 float3 z{ normalize(target - eye) };
302 float3 x{ normalize(cross(up, z)) };
303 float3 y{ cross(z, x) };
304
305 return float4x4{
306 float4{ x.x, y.x, z.x, 0.f },
307 float4{ x.y, y.y, z.y, 0.f },
308 float4{ x.z, y.z, z.z, 0.f },
309 float4{ -dot(x, eye), -dot(y, eye), -dot(z, eye), 1.f }
310 };

327 inline float4x4 float4x4::perspective_lh(const float fov_y_rad, const float aspect, const float z_near,
328 const float z_far) noexcept
329 {
330 const float h{ 1.f / std::tan(fov_y_rad * .5f) };
331 const float w{ h / aspect };
332 const float d{ z_far / (z_far - z_near) };
333
334 return float4x4{
335 float4{ w, 0.f, 0.f, 0.f },
336 float4{ 0.f, h, 0.f, 0.f },
337 float4{ 0.f, 0.f, d, 1.f },
338 float4{ 0.f, 0.f, -d * z_near, 0.f }
339 };
340 }

342 inline float4x4 float4x4::perspective_rh(const float fov_y_rad, const float aspect, const float z_near,
343 const float z_far) noexcept
344 {
345 const float h{ 1.f / std::tan(fov_y_rad * .5f) };
346 const float w{ h / aspect };
347 const float d{ z_far / (z_near - z_far) }; // note sign flip
348
349 return float4x4{

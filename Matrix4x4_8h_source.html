232 inline float4 mul(const float4x4& m, const float4& v) noexcept
233 {
234 return v.x * m.columns[0] +
235 v.y * m.columns[1] +
236 v.z * m.columns[2] +
237 v.w * m.columns[3];
238 }
239
247 inline float4x4 mul(const float4x4& a, const float4x4& b) noexcept
248 {
249 float4x4 result;
250 result.columns[0] = mul(a, b.columns[0]);
251 result.columns[1] = mul(a, b.columns[1]);
252 result.columns[2] = mul(a, b.columns[2]);
253 result.columns[3] = mul(a, b.columns[3]);
254 return result;
255 }
256
260 inline float4 operator*(const float4x4& m, const float4& v) noexcept { return mul(m, v); }
261
265 inline float4x4 operator*(const float4x4& a, const float4x4& b) noexcept { return mul(a, b); }

330 inline float4x4 float4x4::rotate_axis_angle(float3 axis, const float rad) noexcept
331 {
332 axis = normalize(axis);
333 const float c{ cos(rad) };
334 const float s{ sin(rad) };
335 const float t{ 1.0f - c };
336
337 const float x{ axis.x };
338 const float y{ axis.y };
339 const float z{ axis.z };
340
341 return float4x4{
342 float4{ t * x * x + c, t * x * y + s * z, t * x * z - s * y, 0.f },
343 float4{ t * x * y - s * z, t * y * y + c, t * y * z + s * x, 0.f },
344 float4{ t * x * z + s * y, t * y * z - s * x, t * z * z + c, 0.f },
345 float4{ 0.f, 0.f, 0.f, 1.f }
346 };
347 }

349 inline float4x4 float4x4::look_at_lh(const float3 eye, const float3 target, const float3 up) noexcept
350 {

352 float3 x{ normalize(cross(up, z)) };

377 inline float4x4 float4x4::perspective_lh(const float fov_y_rad, const float aspect, const float z_near,
378 const float z_far) noexcept
379 {
380 const float h{ 1.f / tan(fov_y_rad * .5f) };
381 const float w{ h / aspect };
382 const float d{ z_far / (z_far - z_near) };
383
384 return float4x4{
385 float4{ w, 0.f, 0.f, 0.f },
386 float4{ 0.f, h, 0.f, 0.f },
387 float4{ 0.f, 0.f, d, 1.f },
388 float4{ 0.f, 0.f, -d * z_near, 0.f }
389 };

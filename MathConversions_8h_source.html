25 constexpr float4x4 to_float4x4(const quat& q) noexcept
26 {
27 const float xx{ q.x * q.x };
28 const float yy{ q.y * q.y };
29 const float zz{ q.z * q.z };
30 const float xy{ q.x * q.y };
31 const float xz{ q.x * q.z };
32 const float yz{ q.y * q.z };
33 const float wx{ q.w * q.x };
34 const float wy{ q.w * q.y };
35 const float wz{ q.w * q.z };
36
37 return float4x4{
38 float4{ 1.f - 2.f * (yy + zz), 2.f * (xy - wz), 2.f * (xz + wy), 0.f },
39 float4{ 2.f * (xy + wz), 1.f - 2.f * (xx + zz), 2.f * (yz - wx), 0.f },
40 float4{ 2.f * (xz - wy), 2.f * (yz + wx), 1.f - 2.f * (xx + yy), 0.f },

82 constexpr quat quat_from_float3x3(const float3x3& m) noexcept
83 {
84 quat q;
85
86 if (const float trace{ m[0].x + m[1].y + m[2].z }; trace > 0.0f)
87 {
88 const float s{ std::sqrt(trace + 1.0f) * 2.0f };
89 q.w = .25f * s;
90 q.x = (m[2].y - m[1].z) / s;
91 q.y = (m[0].z - m[2].x) / s;
92 q.z = (m[1].x - m[0].y) / s;
93 }
94 else if (m[0].x > m[1].y && m[0].x > m[2].z)
95 {
96 const float s{ std::sqrt(1.f + m[0].x - m[1].y - m[2].z) * 2.f };
97 q.w = (m[2].y - m[1].z) / s;
98 q.x = .25f * s;
99 q.y = (m[0].y + m[1].x) / s;
100 q.z = (m[0].z + m[2].x) / s;
101 }
102 else if (m[1].y > m[2].z)
103 {
104 const float s{ std::sqrt(1.f + m[1].y - m[0].x - m[2].z) * 2.f };
105 q.w = (m[0].z - m[2].x) / s;
106 q.x = (m[0].y + m[1].x) / s;
107 q.y = .25f * s;
108 q.z = (m[1].z + m[2].y) / s;
109 }
110 else
111 {
112 const float s{ std::sqrt(1.f + m[2].z - m[0].x - m[1].y) * 2.f };
113 q.w = (m[1].x - m[0].y) / s;
114 q.x = (m[0].z + m[2].x) / s;
115 q.y = (m[1].z + m[2].y) / s;
116 q.z = .25f * s;
117 }
118
119 return q;
120 }

148 inline float4x4 affine_inverse(const float4x4& m) noexcept
149 {
150 const float3x3 rot{
151 m.columns[0].xyz,
152 m.columns[1].xyz,
153 m.columns[2].xyz
154 };
155
156 float3x3 rot_inv{ inverse_orthonormal(rot) };
157
158 const float3 trans{ m.columns[3].xyz };
159 float3 inv_trans{ -mul(rot_inv, trans) };
160
161 return float4x4{
162 float4{ rot_inv.columns[0].x, rot_inv.columns[0].y, rot_inv.columns[0].z, 0.f },
163 float4{ rot_inv.columns[1].x, rot_inv.columns[1].y, rot_inv.columns[1].z, 0.f },
164 float4{ rot_inv.columns[2].x, rot_inv.columns[2].y, rot_inv.columns[2].z, 0.f },
165 float4{ inv_trans.x, inv_trans.y, inv_trans.z, 1.f }

206 constexpr float4x4 inverse(const float4x4& m) noexcept
207 {
208 // Copy matrix into flat scalar array (column-major -> row-major transpose for row ops)
209 float a[4][4];
210 a[0][0] = m[0].x; a[0][1] = m[1].x; a[0][2] = m[2].x; a[0][3] = m[3].x;
211 a[1][0] = m[0].y; a[1][1] = m[1].y; a[1][2] = m[2].y; a[1][3] = m[3].y;
212 a[2][0] = m[0].z; a[2][1] = m[1].z; a[2][2] = m[2].z; a[2][3] = m[3].z;
213 a[3][0] = m[0].w; a[3][1] = m[1].w; a[3][2] = m[2].w; a[3][3] = m[3].w;
214
215 // Augmented identity
216 float inv[4][4]{
217 {1, 0, 0, 0},
218 {0, 1, 0, 0},
219 {0, 0, 1, 0},
220 {0, 0, 0, 1}
221 };
222
223 for (int i{ 0 }; i < 4; ++i)
224 {
225 // Partial pivoting
226 int pivot{ i };
227 float max_val{ std::abs(a[i][i]) };
228 for (int j = i + 1; j < 4; ++j)
229 {
230 if (const float val{ std::abs(a[j][i]) }; val > max_val)
231 {
232 max_val = val;
233 pivot = j;
234 }
235 }
236
237 if (max_val < epsilon)
238 return float4x4::identity(); // Singular
239
240 // Swap rows
241 if (pivot != i)
242 {
243 for (int k{ 0 }; k < 4; ++k)
244 {
245 std::swap(a[i][k], a[pivot][k]);
246 std::swap(inv[i][k], inv[pivot][k]);
247 }
248 }
249
250 // Normalize pivot row
251 const float inv_pivot{ 1.0f / a[i][i] };
252 for (int k{ 0 }; k < 4; ++k)
253 {
254 a[i][k] *= inv_pivot;
255 inv[i][k] *= inv_pivot;
256 }
257
258 // Eliminate column
259 for (int j{ 0 }; j < 4; ++j)
260 {
261 if (j == i) continue;
262 const float factor{ a[j][i] };
263 for (int k{ 0 }; k < 4; ++k)
264 {
265 a[j][k] -= a[i][k] * factor;
266 inv[j][k] -= inv[i][k] * factor;
267 }
268 }
269 }
270
271 // Transpose inv back into column-major float4x4
272 return {
273 float4{ inv[0][0], inv[1][0], inv[2][0], inv[3][0] },
274 float4{ inv[0][1], inv[1][1], inv[2][1], inv[3][1] },
275 float4{ inv[0][2], inv[1][2], inv[2][2], inv[3][2] },
276 float4{ inv[0][3], inv[1][3], inv[2][3], inv[3][3] }
277 };
278 }

54 inline quat quat_from_euler(const float yaw_z, const float pitch_x, const float roll_y) noexcept
55 {
56 const float cy{ std::cos(yaw_z * .5f) };
57 const float sy{ std::sin(yaw_z * .5f) };
58 const float cp{ std::cos(pitch_x * .5f) };
59 const float sp{ std::sin(pitch_x * .5f) };
60 const float cr{ std::cos(roll_y * .5f) };
61 const float sr{ std::sin(roll_y * .5f) };
62
63 return quat{
64 sr * cp * cy - cr * sp * sy, // x
65 cr * sp * cy + sr * cp * sy, // y
66 cr * cp * sy - sr * sp * cy, // z
67 cr * cp * cy + sr * sp * sy // w
68 };
69 }

80 inline quat mul(const quat& a, const quat& b) noexcept
81 {
82 return quat{
83 a.w * b.x + a.x * b.w + a.y * b.z - a.z * b.y,
84 a.w * b.y - a.x * b.z + a.y * b.w + a.z * b.x,
85 a.w * b.z + a.x * b.y - a.y * b.x + a.z * b.w,
86 a.w * b.w - a.x * b.x - a.y * b.y - a.z * b.z
87 };

111 inline quat inverse(const quat& q) noexcept
112 {
113 const float len_sq{ dot(q, q) };
114
115 if (almost_equal(len_sq, .0f)) return quat_identity();
116
117 const quat conj{ conjugate(q) };
118 return conj * (1.f / len_sq);
119 }

148 inline quat slerp(const quat& a, const quat& b, const float t) noexcept
149 {
150 float d{ dot(a, b) };
151 quat b_adj{ b };
152
153 // If dot is negative, slerp won't take shortest path - flip one
154 if (d < 0.f)
155 {
156 b_adj = -b;
157 d = -d;
158 }
159
160 // If quats are very close, fall back to nlerp
161 if (d > .9995f)
162 {
163 return nlerp(a, b_adj, t);
164 }
165
166 const float theta{ std::acos(clamp(d, -1.f, 1.f)) };
167 const float sine{ std::sin(theta) };
168
169 const float wa{ std::sin((1.f - t) * theta) / sine };
170 const float wb{ std::sin(t * theta) / sine };
171
172 return a * wa + b_adj * wb;
173 }

185 inline float3 rotate_vector(const quat& q, float3 v) noexcept
186 {

66 static float3x3 quat_to_float3x3_internal(const quat& q) noexcept
67 {
68 const float xx{ q.x * q.x };
69 const float yy{ q.y * q.y };
70 const float zz{ q.z * q.z };
71 const float xy{ q.x * q.y };
72 const float xz{ q.x * q.z };
73 const float yz{ q.y * q.z };
74 const float wx{ q.w * q.x };
75 const float wy{ q.w * q.y };
76 const float wz{ q.w * q.z };
77
78 return float3x3{
79 float3{ 1.f - 2.f * (yy + zz), 2.f * (xy - wz), 2.f * (xz + wy) },
80 float3{ 2.f * (xy + wz), 1.f - 2.f * (xx + zz), 2.f * (yz - wx) },
81 float3{ 2.f * (xz - wy), 2.f * (yz + wx), 1.f - 2.f * (xx + yy) }
82 };
83 }

134 inline float3 mul(const float3x3& m, const float3& v) noexcept
135 {
136 return v.x * m.columns[0] +
137 v.y * m.columns[1] +
138 v.z * m.columns[2];
139 }

150 inline float3x3 mul(const float3x3& a, const float3x3& b) noexcept
151 {
152 float3x3 result;
153 result.columns[0] = mul(a, b.columns[0]);
154 result.columns[1] = mul(a, b.columns[1]);
155 result.columns[2] = mul(a, b.columns[2]);
156 return result;
157 }
